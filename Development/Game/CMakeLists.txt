# [[Defaults]]
set(PROJECT_GAME_MODULE_NAME "${PROJECT_NAME}")


# [[Binaries]]
eightgine_add_module(MODULE_NAME "${PROJECT_GAME_MODULE_NAME}"
    MODULE_SOURCES_DIR "${CMAKE_CURRENT_LIST_DIR}"
    MODULE_INCLUDE_DIR "${CMAKE_CURRENT_LIST_DIR}"
)

eightgine_add_dependency(MODULE_OR_EXECUTABLE_NAME "${PROJECT_GAME_MODULE_NAME}"
    DEPENDENCY_NAME "EightgineCore"
)
