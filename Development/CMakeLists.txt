# # Functions
function(copy_shared_libraries SHARED_LIBRARIES_PATH)
    foreach(SHARED_LIBRARY_NAME IN LISTS ARGN)
        set(src "${SHARED_LIBRARIES_PATH}/${SHARED_LIBRARY_NAME}.dll")
        set(dst "$<TARGET_FILE_DIR:${PROJECT_NAME}>/${SHARED_LIBRARY_NAME}.dll")
        add_custom_command(TARGET ${PROJECT_NAME} POST_BUILD COMMAND ${CMAKE_COMMAND} -E copy_if_different "${src}" "${dst}")
    endforeach()
endfunction()

function(add_third_party_libraries THIRD_PARTY_DIR)
    set(THIRD_PARTY_LIBRARY_NAMES ${ARGN})
    foreach(THIRD_PARTY_LIBRARY_NAME IN LISTS THIRD_PARTY_LIBRARY_NAMES)
        target_include_directories(${EIGHTGINE_LIBRARY_NAME} PUBLIC "${THIRD_PARTY_DIR}/${THIRD_PARTY_LIBRARY_NAME}/include")
        target_link_directories(${EIGHTGINE_LIBRARY_NAME} PUBLIC "${THIRD_PARTY_DIR}/${THIRD_PARTY_LIBRARY_NAME}/lib")
        target_link_libraries(${EIGHTGINE_LIBRARY_NAME} PUBLIC ${THIRD_PARTY_LIBRARY_NAME})
        copy_shared_libraries("${THIRD_PARTY_DIR}/${THIRD_PARTY_LIBRARY_NAME}/bin" "${THIRD_PARTY_LIBRARY_NAME}")
    endforeach()
endfunction()
